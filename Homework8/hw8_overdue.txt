int sensorPin = A0;    // select the input pin for the potentiometerint ledPin = 5;      // select the pin for the LEDint sensorValue = 0;  // variable to store the value coming from the sensorint delayamount;void setup() {  // declare the ledPin as an OUTPUT:  pinMode(ledPin, OUTPUT);  Serial.begin(9600);}void loop() {  // read the value from the sensor:  sensorValue = analogRead(sensorPin);  delayamount = map(sensorValue,0,1023,20,1000);  Serial.println(sensorValue);  // turn the ledPin on  digitalWrite(ledPin, HIGH);  // stop the program for <sensorValue> milliseconds:  delay(delayamount);  // turn the ledPin off:  digitalWrite(ledPin, LOW);  // stop the program for for <sensorValue> milliseconds:  delay(delayamount);}
