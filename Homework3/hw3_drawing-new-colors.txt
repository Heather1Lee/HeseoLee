function setup() {   createCanvas(400, 400);} var diameter = 10;function draw() {   if (mouseIsPressed) {    ellipse(mouseX, mouseY, diameter);  }}function keyPressed() {  print(key);  if (key == 'R') {    fill(255, 0, 0);  } else if (key == 'G') {    fill(200, 194, 189);  } else if (key == 'B') {    fill(128, 66, 181);   } else if (key == 'Q') {    fill(255, 89, 143);   } else if (key == 'W') {    fill(255, 215, 0);   } else if (key == 'E') {    fill(89, 173, 208);   }}
